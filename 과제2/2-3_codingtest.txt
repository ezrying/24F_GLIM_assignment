#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> num_list, int n) {
    // 인덱스 0~n-1 & n부터 num_list.size()로 나눈 배열을 만든 뒤 순서 바꾸고 합쳐서 return;
    vector<int> vec1;
    vector<int> vec2;
    for(int i=0;i<n;++i){
        vec1.push_back(num_list[i]);
    }
    for(int i=n;i<num_list.size();++i){
        vec2.push_back(num_list[i]);
    }
    vec2.insert(vec2.end(),vec1.begin(),vec1.end());
    return vec2;
}
